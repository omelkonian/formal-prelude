macro
  ¡_ : Term → Tactic
  (¡ f) hole = do
    ty ← normalise =<< inferType f
    let n = (length ∘ vArgs ∘ proj₁ ∘ viewTy′) ty
    f′ ← f -∗- L.replicate n (quote auto ∙)
    unify hole f′

  ‼_ : Name → Tactic
  (‼ f) hole = do
    ty ← normalise =<< getType f
    let n = length $ vArgs $ viewTy′ ty .proj₁
    let as = L.replicate n (vArg $ quote auto ∙)
    unify hole $ con f as
